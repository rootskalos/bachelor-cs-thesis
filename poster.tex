% --- 1. PROBLEMA ---
\begin{pblock}{1}{Problema y contexto clĂ­nico}
\textbf{\color{udark}Glioblastoma multiforme (GBM):} tumor cerebral primario maligno mĂ¡s frecuente y agresivo en adultos.
\begin{itemize}\itemsep2pt
  \item Mediana de supervivencia \textbf{12--15 meses}; supervivencia a 5 aĂ±os \textbf{$<$7\,\%}.
  \item EstĂ¡ndar (protocolo de Stupp): cirugĂ­a + radioterapia + \textbf{temozolomida} (TMZ). Casi todos los pacientes recaen con \emph{resistencia a la TMZ}.
\end{itemize}

\textbf{\color{udark}La recaĂ­da no es accidental, es evolutiva.} La dosis mĂ¡xima tolerada (MTD) ejerce una presiĂ³n selectiva que \textbf{expande los subclones resistentes}: al eliminar las cĂ©lulas sensibles, libera a las resistentes de la competencia.

\vspace{2mm}
\textbf{\color{uneg}$\blacktriangle$\ Brecha que aborda este trabajo}
\begin{itemize}\itemsep2pt
  \item \textbf{RL de agente Ăºnico:} trata al tumor como una dinĂ¡mica \emph{fija}; no anticipa ni responde estratĂ©gicamente.
  \item \textbf{HeurĂ­sticas adaptativas (Gatenby):} reglas fijas encendido/apagado, \emph{no optimizadas}.
  \item \textbf{Modelos teĂ³rico-de-juegos:} resuelven al tumor analĂ­ticamente $\Rightarrow$ pobreza de comportamientos adversariales.
  \item \textbf{MĂ©tricas de carga aislada:} premian la reducciĂ³n agresiva del tumor $\Rightarrow$ \emph{inducen} resistencia (engaĂ±osas).
\end{itemize}

% --- 2. OBJETIVOS ---
\begin{pblock}{2}{Objetivos}
\textbf{\color{udark}General.} DiseĂ±ar \textbf{GBMARL}, un marco de aprendizaje por refuerzo multiagente adversarial que modele la terapia y la resistencia del glioblastoma como agentes en competencia sobre un simulador calibrado con datos reales, y evaluar su capacidad de \textbf{retrasar la intratabilidad} frente a lĂ­neas base clĂ­nicas.

\vspace{2mm}
\textbf{\color{udark}EspecĂ­ficos.}
\begin{enumerate}\itemsep1pt
  \item Construir y calibrar un \textbf{simulador de EDO} (competencia Lotka--Volterra + farmacodinĂ¡mica tipo Hill) con datos reales (GDSC2 \& DepMap).
  \item Formular el \textbf{entorno multiagente} con observabilidad parcial clĂ­nica e implementar \textbf{MAPPO-CTDE} y su ablaciĂ³n \textbf{IPPO}.
  \item Entrenar y validar con \textbf{mĂºltiples semillas}, reportando significancia estadĂ­stica e \textbf{interpretando el mecanismo} de la polĂ­tica.
  \item Aislar la contribuciĂ³n de CTDE (\textbf{ablaciĂ³n}) y evaluar la \textbf{sensibilidad} a las constantes de calibraciĂ³n.

% --- 3. METODOLOGIA ---
\begin{pblock}{3}{MetodologĂ­a --- el marco GBMARL}
\textbf{\color{udark}Juego adversarial de suma no nula.} Dos agentes en \emph{self-play} sobre un simulador de EDO:
\begin{itemize}\itemsep1pt
  \item \textbf{Agente de terapia} elige la dosis $u(t)\in[0,u_{\max}]$ $\to$ maximizar los dĂ­as con tumor manejable.
  \item \textbf{Agente tumoral} elige la transiciĂ³n fenotĂ­pica $\phi\in[0,\phi_{\max}]$ $\to$ forzar la dominancia resistente.
\end{itemize}

\textbf{\color{udark}Simulador calibrado} (Runge--Kutta 4):
\[
\small
\begin{aligned}
\tfrac{dS}{dt} &= \alpha_S S\!\left(1-\tfrac{S+R}{K}\right)-\delta_S(c)S-\phi S\\
\tfrac{dR}{dt} &= \alpha_R R\!\left(1-\tfrac{S+R}{K}\right)-\delta_R(c)R+\phi S\\
\tfrac{dc}{dt} &= -\lambda_c c+u(t),\qquad \delta_X(c)=\delta_X^{\max}\tfrac{c^{h}}{\mathrm{IC50}_X^{h}+c^{h}}
\end{aligned}
\]
\bitem \textbf{CalibraciĂ³n:} datos GDSC2+DepMap fijan la \emph{potencia} del fĂ¡rmaco; la literatura ancla el techo de muerte. $\mathrm{IC50}_S{=}0.36$, $\mathrm{IC50}_R{=}3.27$ $\Rightarrow$ \textbf{brecha $\approx 9\times$} (34 lĂ­neas recuperadas).

\bitem \textbf{Observabilidad parcial clĂ­nica:} la terapia observa $(S{+}R,\,c)$ pero \textbf{no} la fracciĂ³n resistente $\Rightarrow$ justifica un \textbf{crĂ­tico centralizado} $V(s^{\text{global}})$.

\bitem \textbf{MAPPO-CTDE:} actor local + crĂ­tico con estado conjunto; ventajas GAE; \emph{self-play} adversarial.

\bitem \textbf{MĂ©trica honesta (TTP-combinado):} dĂ­as hasta fallar \emph{control} de carga \textbf{o} \emph{tratable} (mayorĂ­a resistente), reportando el \textbf{modo de falla}.
